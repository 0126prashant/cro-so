function func(n,arr){
// 	console.log(n,arr)
	let lastOcc = {}
	for(i=n-1;i>=0;i--){
		if(!(arr[i] in lastOcc)){
			lastOcc[arr[i]]=i
		}
		
		
	}
	let moves =0
		let currPos = 0
		while(currPos < n-1){
			let nextPos = lastOcc[arr[currPos]]
			currPos = nextPos
			moves+=1
		}
	return moves
	
}

function func(n,k,arr){
// 	console.log(n,k,arr)
	if(n%k!==0){
		return -1
	}
	arr.sort((a,b) => a-b)
// 	console.log(arr)
	
	let teamnum = n/k
	let teams = []
	for(let i=0;i<teamnum;i++){
		let team = arr.splice(i*k,(i+1)*k)
		const maxscore = Math.max(...team)
		const minscore = Math.min(...team)
		const noncom = maxscore-minscore
		
		if(noncom==0 || team.length<k){
			return -1
		}
		team.push(noncom)
	}
	return team.reduce((total,score)=> total +score,0)
}
